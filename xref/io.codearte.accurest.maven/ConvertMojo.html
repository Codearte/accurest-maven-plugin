22  /**
23   *
24   * Convert Accurest contracts into WireMock stubs mappings.
25   *
26   * This goal allow to generate `stubs-jar` or execute `accurest:run` with generated WireMock mappings.
27   *
28   */
29  @Mojo(name = "convert", requiresProject = false, defaultPhase = LifecyclePhase.PROCESS_TEST_RESOURCES)
30  @CompileStatic
31  public class ConvertMojo extends AbstractMojo {
32  
33      /**
34       * Directory containing Accurest contracts written using the GroovyDSL
35       */
36      @Parameter(defaultValue = "${basedir}/src/test/resources/accurest")
37      private File contractsDirectory;
38  
39      /**
40       * Directory where the generated WireMock stubs from Groovy DSL should be placed.
41       * You can then mention them in your packaging task to create jar with stubs
42       */
43      @Parameter(defaultValue = "${project.build.directory}/accurest")
44      private File outputDirectory;
45  
46      /**
47       * Directory containing contracts written using the GroovyDSL
48       *
49       * This parameter is only used when goal is executed outside of maven project.
50       */
51      @Parameter(property = "contractsDirectory", defaultValue = "${basedir}")
52      private File source;
53  
54      @Parameter(property = "stubsDirectory", defaultValue = "${basedir}")
55      private File destination;
56  
57      @Parameter(property = "accurest.skip", defaultValue = "false")
58      private boolean skip;
59  
60      @Parameter(defaultValue = "${session}", readonly = true)
61      private MavenSession mavenSession;
62  
63      @Parameter(defaultValue = "${project}", readonly = true)
64      private MavenProject project;
65  
66      @Component(role = MavenResourcesFiltering.class, hint = "default")
67      private MavenResourcesFiltering mavenResourcesFiltering;
68  
69      public void execute() throws MojoExecutionException, MojoFailureException {
70  
71          if (skip) {
72              getLog().info(String.format("Skipping accurest execution: accurest.skip=%s", skip));
73              return;
74          }
75  
76          new CopyContracts(project, mavenSession, mavenResourcesFiltering).copy(contractsDirectory, outputDirectory);
77  
78          final AccurestConfigProperties config = new AccurestConfigProperties();
79          config.setContractsDslDir(isInsideProject() ? contractsDirectory : source);
80          config.setStubsOutputDir(isInsideProject() ? new File(outputDirectory, "mappings") : destination);
81  
82          getLog().info("Converting from accurest contracts written in GroovyDSL to WireMock stubs mappings");
83          getLog().info(String.format("     Accurest contracts directory: %s", config.getContractsDslDir()));
84          getLog().info(String.format("WireMock stubs mappings directory: %s", config.getStubsOutputDir()));
85  
86          RecursiveFilesConverter converter = new RecursiveFilesConverter(new DslToWireMockClientConverter(), config);
87          converter.processFiles();
88      }
89  
90      private boolean isInsideProject() {
91          return mavenSession.getRequest().isProjectPresent();
92      }
93  
94  
95  }
