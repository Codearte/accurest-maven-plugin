1   package io.codearte.accurest.maven;
2   
3   import static java.lang.String.format;
4   
5   import java.io.File;
6   
7   import org.apache.maven.plugin.AbstractMojo;
8   import org.apache.maven.plugin.MojoExecutionException;
9   import org.apache.maven.plugin.MojoFailureException;
10  import org.apache.maven.plugins.annotations.LifecyclePhase;
11  import org.apache.maven.plugins.annotations.Mojo;
12  import org.apache.maven.plugins.annotations.Parameter;
13  
14  import io.codearte.accurest.AccurestException;
15  import io.codearte.accurest.TestGenerator;
16  import io.codearte.accurest.config.AccurestConfigProperties;
17  import io.codearte.accurest.config.TestFramework;
18  import io.codearte.accurest.config.TestMode;
19  
20  @Mojo(name = "generateTests", defaultPhase = LifecyclePhase.GENERATE_TEST_SOURCES)
21  public class GenerateTestsMojo extends AbstractMojo {
22  
23  	@Parameter(defaultValue = "${basedir}", readonly = true, required = true)
24  	private File baseDir;
25  
26  	@Parameter(defaultValue = "${project.build.directory}", readonly = true, required = true)
27  	private File projectBuildDirectory;
28  
29  	@Parameter(defaultValue = "/src/test/accurest")
30  	private String contractsDir;
31  
32  	@Parameter(defaultValue = "/generated-test-sources/accurest")
33  	private String generatedTestSourcesDir;
34  
35  	@Parameter(defaultValue = "io.codearte.accurest.tests")
36  	private String basePackageForTests;
37  
38  	@Parameter
39  	private String baseClassForTests;
40  
41  	@Parameter(defaultValue = "MOCKMVC")
42  	private TestMode testMode;
43  
44  	@Parameter(defaultValue = "JUNIT")
45  	private TestFramework testFramework;
46  
47  	@Parameter
48  	private String ruleClassForTests;
49  
50  	@Parameter
51  	private String nameSuffixForTests;
52  
53  	public void execute() throws MojoExecutionException, MojoFailureException {
54  		getLog().info("Generating server tests source code for Accurest contract verification");
55  
56  		AccurestConfigProperties config = new AccurestConfigProperties();
57  		config.setContractsDslDir(new File(baseDir, contractsDir));
58  		config.setGeneratedTestSourcesDir(new File(projectBuildDirectory, generatedTestSourcesDir));
59  		config.setTargetFramework(testFramework);
60  		config.setTestMode(testMode);
61  		config.setBasePackageForTests(basePackageForTests);
62  		config.setBaseClassForTests(baseClassForTests);
63  		config.setRuleClassForTests(ruleClassForTests);
64  		config.setNameSuffixForTests(nameSuffixForTests);
65  
66  		getLog().info(format("Using %s as test source directory", config.getGeneratedTestSourcesDir()));
67  		getLog().info(format("Using %s as base class for test classes", config.getBaseClassForTests()));
68  
69  		try {
70  			TestGenerator generator = new TestGenerator(config);
71  			int generatedClasses = generator.generate();
72  			getLog().info(format("Generated %s test classes.", generatedClasses));
73  		} catch (AccurestException e) {
74  			throw new MojoExecutionException(format("Accurest Plugin exception: %s", e.getMessage()), e);
75  		}
76  	}
77  
78  }
