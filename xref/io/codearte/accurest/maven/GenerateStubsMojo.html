7   import org.apache.maven.plugin.AbstractMojo;

10  import org.apache.maven.plugins.annotations.LifecyclePhase;
11  import org.apache.maven.plugins.annotations.Mojo;
12  import org.apache.maven.plugins.annotations.Parameter;
13  
14  import io.codearte.accurest.config.AccurestConfigProperties;
15  import io.codearte.accurest.wiremock.DslToWireMockClientConverter;
16  import io.codearte.accurest.wiremock.RecursiveFilesConverter;
17  
18  @Mojo(name = "generateStubs", defaultPhase = LifecyclePhase.PROCESS_RESOURCES)
19  public class GenerateStubsMojo extends AbstractMojo {
20  
21  	@Parameter(defaultValue = "${basedir}", readonly = true, required = true)
22  	private File baseDir;
23  
24  	@Parameter(defaultValue = "${project.build.directory}", readonly = true, required = true)
25  	private File projectBuildDirectory;
26  
27  	@Parameter(defaultValue = "/src/test/accurest")
28  	private String contractsDir;
29  
30  	@Parameter(defaultValue = "mappings")
31  	private String mappingsDir;
32  
33  	public void execute() throws MojoExecutionException, MojoFailureException {
34  		AccurestConfigProperties config = new AccurestConfigProperties();
35  		config.setContractsDslDir(new File(baseDir, contractsDir));
36  		config.setStubsOutputDir(new File(projectBuildDirectory, mappingsDir));
37  
38  		getLog().info("Accurest Plugin: Invoking GroovyDSL to WireMock client stubs conversion");
39  		getLog().info(format("From '%s' to '%s'", config.getContractsDslDir(), config.getStubsOutputDir()));
40  
41  		RecursiveFilesConverter converter = new RecursiveFilesConverter(new DslToWireMockClientConverter(), config);
42  		converter.processFiles();
43  	}
44  
45  }
